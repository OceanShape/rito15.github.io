
# 구조체
---

## Vertex Input

- 입력 조립기 -> 버텍스 쉐이더

```hlsl
struct vi
{
   //Type  (Name)   Semantic

    float4 vertex : POSITION  ; // 로컬 스페이스 정점 위치
    float3 normal : NORMAL    ; // 로컬 스페이스 정점 노멀
    float4 uv     : TEXCOORD0 ; // 버텍스 UV - float2로 사용해도 됨
                                // TEXCOORD[n]
    float4 tangent: TANGENT   ; // 메시에서 계산되거나 임포트된 탄젠트 벡터
    float4 color  : COLOR     ; // 버텍스 컬러
};
```

<br>

## Vertex Output(Vertex to Fragment)

- 버텍스 쉐이더 -> 프래그먼트 쉐이더

```hlsl
struct v2f
{
    float4 vertex : SV_POSITION ; // SV : System Value
    float3 normal : NORMAL      ;
    float2 uv     : TEXCOORD0   ;
};
```

## Vertex Position

```hlsl
struct vi
{
    float4 vertex: POSITION;
};
struct v2f
{
    float4 vertex: SV_POSITION; // SV : System Value
};
v2f vert(vi v)
{
    v2f o;
    o.vertex = TransformObjectToHClip(v.vertex.xyz);
}
```

<br>

## Vertex Normal

```hlsl
struct vi
{
    float3 normal: NORMAL;
};
struct v2f
{
    float3 worldNormal: NORMAL;
};
v2f vert(vi v)
{
    v2f o;
    o.worldNormal = TransformObjectToWorldNormal(v.normal);
}
```

<br>

## Texture Coordinates(UV)

```hlsl
struct vi
{
    float2 uv: TEXCOORD0;
};
struct v2f
{
    float2 uv: TEXCOORD0;
};
v2f vert(vi v)
{
    v2f o;
    o.uv = v.uv;
}
```
